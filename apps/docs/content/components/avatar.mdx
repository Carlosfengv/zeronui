## Usage

```tsx
import { Avatar, AvatarFallback, AvatarImage } from "@zeron-ui/ui/avatar";

export function Example() {
  return (
    <Avatar>
      <AvatarImage alt="User" src="/avatars/shadcn.jpg" />
      <AvatarFallback>UI</AvatarFallback>
    </Avatar>
  );
}
```
