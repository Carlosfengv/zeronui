## Usage

```tsx
import { Kbd, KbdGroup } from "@zeron-ui/ui/kbd";

export function Example() {
  return (
    <KbdGroup>
      <Kbd>⌘</Kbd>
      <Kbd>K</Kbd>
    </KbdGroup>
  );
}
```
