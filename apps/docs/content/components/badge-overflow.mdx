## Usage

```tsx
import { Badge } from "@zeron-ui/ui/badge";
import { BadgeOverflow } from "@zeron-ui/ui/badge-overflow";

export function Example() {
  return (
    <BadgeOverflow
      items={["React", "TypeScript", "Next.js"]}
      renderBadge={(_, label) => <Badge>{label}</Badge>}
    />
  );
}
```

## Object Items

```tsx
<BadgeOverflow
  items={[
    { id: 1, name: "React" },
    { id: 2, name: "TypeScript" },
  ]}
  getBadgeLabel={(item) => item.name}
  getBadgeKey={(item) => item.id}
  renderBadge={(_, label) => <Badge>{label}</Badge>}
/>
```
