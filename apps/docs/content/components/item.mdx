## Usage

```tsx
import {
  Item,
  ItemContent,
  ItemDescription,
  ItemTitle,
} from "@zeron-ui/ui/item";

export function Example() {
  return (
    <Item>
      <ItemContent>
        <ItemTitle>Title</ItemTitle>
        <ItemDescription>Description</ItemDescription>
      </ItemContent>
    </Item>
  );
}
```
