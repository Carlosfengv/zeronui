## Usage

```tsx
import { Field, FieldDescription, FieldLabel } from "@zeron-ui/ui/field";
import { Input } from "@zeron-ui/ui/input";

export function Example() {
  return (
    <Field>
      <FieldLabel htmlFor="email">Email</FieldLabel>
      <Input id="email" placeholder="hello@example.com" />
      <FieldDescription>Used for account notifications.</FieldDescription>
    </Field>
  );
}
```
