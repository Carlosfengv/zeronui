## Usage

```tsx
import { CheckboxGroup, CheckboxGroupItem } from "@zeron-ui/ui/checkbox-group";

export function Example() {
  return (
    <CheckboxGroup>
      <CheckboxGroupItem id="one">Option One</CheckboxGroupItem>
      <CheckboxGroupItem id="two">Option Two</CheckboxGroupItem>
    </CheckboxGroup>
  );
}
```
