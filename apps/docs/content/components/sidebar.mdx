## Usage

```tsx
import {
  Sidebar,
  SidebarContent,
  SidebarProvider,
  SidebarTrigger,
} from "@zeron-ui/ui/sidebar";

export function Example() {
  return (
    <SidebarProvider>
      <Sidebar>
        <SidebarContent />
      </Sidebar>
      <SidebarTrigger />
    </SidebarProvider>
  );
}
```
