## Usage

```tsx
import {
  Stepper,
  StepperContent,
  StepperIndicator,
  StepperItem,
  StepperList,
  StepperTrigger,
} from "@zeron-ui/ui/stepper";

export function Example() {
  return (
    <Stepper defaultValue="account">
      <StepperList>
        <StepperItem value="account">
          <StepperTrigger>
            <StepperIndicator />
          </StepperTrigger>
        </StepperItem>
        <StepperItem value="profile">
          <StepperTrigger>
            <StepperIndicator />
          </StepperTrigger>
        </StepperItem>
      </StepperList>
      <StepperContent value="account">Account setup</StepperContent>
      <StepperContent value="profile">Profile details</StepperContent>
    </Stepper>
  );
}
```

## Validation

```tsx
<Stepper
  defaultValue="account"
  onValidate={async (_value, direction) => {
    return direction === "prev" || formIsValid;
  }}
>
  {/* steps */}
</Stepper>
```
