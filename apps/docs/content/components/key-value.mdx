## Usage

```tsx
import { KeyValue, type KeyValueItem } from "@zeron-ui/ui/key-value";

const items: KeyValueItem[] = [
  {
    id: "database-url",
    key: "DATABASE_URL",
    masked: true,
    value: "postgres://production.internal/app",
  },
];

export function Example() {
  return <KeyValue editing items={items} onItemsChange={() => {}} />;
}
```
