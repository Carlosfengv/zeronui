## Usage

```tsx
import {
  InputGroup,
  InputGroupAddon,
  InputGroupInput,
} from "@zeron-ui/ui/input-group";

export function Example() {
  return (
    <InputGroup>
      <InputGroupInput placeholder="Search..." />
      <InputGroupAddon>⌘K</InputGroupAddon>
    </InputGroup>
  );
}
```
